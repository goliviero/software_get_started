{

    \parbox[t][\paperheight]{\paperwidth}
    {
      \hfill
      \colorbox{blue}
      {
        \begin{minipage}[b]{3em}
          \centering\Huge\textcolor{white}{T\\H\\E\\S\\E\\}
          \vspace{0.3cm}
        \end{minipage}
      }
    }
  }
}
